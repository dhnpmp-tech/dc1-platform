## Install

```bash
npm install dc1-renter-sdk
```

## Run a complete job flow

```ts
import { DC1RenterClient } from 'dc1-renter-sdk'

const baseUrl = 'https://api.dcp.sa'
const apiKey = process.env.DC1_RENTER_KEY!

const client = new DC1RenterClient({ apiKey, baseUrl })

// 1) Validate auth
const me = await client.me()
console.log('renter:', me.email)

// 2) Submit job
const submitted = await client.submitJob({
  job_type: 'llm-inference',
  duration_minutes: 3,
  max_duration_seconds: 180,
  container_spec: { image_type: 'vllm-serve' },
  params: {
    model: 'TinyLlama/TinyLlama-1.1B-Chat-v1.0',
    prompt: 'Explain GPU marketplaces in 3 bullets.',
  },
})

const jobId = submitted.job_id
console.log('job_id:', jobId)

// 3) Stream logs (SSE)
const logRes = await fetch(`${baseUrl}/api/jobs/${jobId}/logs/stream?key=${apiKey}`)
if (!logRes.ok || !logRes.body) throw new Error('Failed to open log stream')

const reader = logRes.body.getReader()
const decoder = new TextDecoder()
let logDone = false

while (!logDone) {
  const { value, done } = await reader.read()
  if (done) break
  const chunk = decoder.decode(value, { stream: true })
  for (const line of chunk.split('\n')) {
    if (!line.startsWith('data: ')) continue
    const evt = JSON.parse(line.slice(6))
    if (evt.type === 'log') console.log('[log]', evt.line)
    if (evt.type === 'end') {
      console.log('[end]', evt.status)
      logDone = true
      break
    }
  }
}

// 4) Fetch output
const outputRes = await fetch(`${baseUrl}/api/jobs/${jobId}/output?key=${apiKey}`)
const output = await outputRes.json()
console.log('output:', output)
```

## Notes

- `duration_minutes` and `container_spec.image_type` are required by `POST /api/jobs/submit`.
- Log streaming endpoint: `GET /api/jobs/:job_id/logs/stream`.
- Output endpoint: `GET /api/jobs/:job_id/output`.
